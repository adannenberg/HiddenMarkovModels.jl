julia> function random_gaussian_hmm(N)
           p, A = ones(N) / N, rand_trans_mat(N)
           μ, σ = randn(N), ones(N)
           dists = [Normal(μ[n], σ[n]) for n in 1:N]
           return HMM(p, A, dists)
       end;
